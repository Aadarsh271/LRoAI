lroai_country_current_treasury = {
    if = {
        limit = {
            scaled_debt <= 0
        }
        value = gold_reserves
    }
    else = {
        value = 0
        subtract = scaled_debt
        multiply = credit
    }
}

lroai_95_percent_of_credit = {
    value = 0
    subtract = credit
    multiply = 0.95
}

lroai_country_budget_surplus = {
    if = {
        limit = {
            has_variable = lroai_country_budget_surplus
            var:lroai_country_budget_surplus != 0
        }
        value = var:lroai_country_budget_surplus
        if = {
            limit = {
                var:lroai_country_budget_surplus < 0
            }
            ceiling = yes
        }
        else = {
            floor = yes
        }
        divide = {
            if = {
                limit = {
                    var:lroai_country_budget_surplus < 0
                }
                value = 0
                subtract = {
                    value = var:lroai_country_budget_surplus
                    subtract = {
                        value = var:lroai_country_budget_surplus
                        ceiling = yes
                    }
                }
            }
            else = {
                value = var:lroai_country_budget_surplus
                subtract = {
                    value = var:lroai_country_budget_surplus
                    floor = yes
                }
            }
            multiply = 100
        }
    }
    else = {
        value = 0
    }
}

lroai_country_fixed_income = {
    value = income
    if = {
        limit = {
            has_variable = lroai_investment_pool_transfer
            var:lroai_investment_pool_transfer > 0
        }
        subtract = lroai_investment_pool_transfer
    }
    if = {
        limit = {
            is_player = no
        }
        if = {
            limit = {
                tax_level = very_low
            }
            multiply = lroai_very_low_tax_level_factor
        }
        else_if = {
            limit = {
                tax_level = low
            }
            multiply = lroai_low_tax_level_factor
        }
        else_if = {
            limit = {
                tax_level = high
            }
            multiply = lroai_high_tax_level_factor
        }
        else_if = {
            limit = {
                tax_level = very_high
            }
            multiply = lroai_very_high_tax_level_factor
        }
    }
}

lroai_country_active_income = {
    if = {
        limit = {
            has_variable = lroai_country_budget_surplus
            var:lroai_country_budget_surplus != 0
            has_variable = lroai_total_building_expenses
            var:lroai_total_building_expenses > 0
            has_variable = lroai_investment_pool_transfer
            var:lroai_investment_pool_transfer > 0
        }
        value = lroai_country_budget_surplus
        add = lroai_total_building_expenses
        subtract = lroai_investment_pool_transfer
    }
    else = {
        value = lroai_country_fixed_income
    }
    if = {
        limit = {
            is_player = no
        }
        if = {
            limit = {
                tax_level = very_low
            }
            multiply = lroai_very_low_tax_level_factor
        }
        else_if = {
            limit = {
                tax_level = low
            }
            multiply = lroai_low_tax_level_factor
        }
        else_if = {
            limit = {
                tax_level = high
            }
            multiply = lroai_high_tax_level_factor
        }
        else_if = {
            limit = {
                tax_level = very_high
            }
            multiply = lroai_very_high_tax_level_factor
        }
    }
    min = 1
}

lroai_investment_pool_potential = {
    value = lroai_investment_pool_transfer
    if = {
        limit = {
            lroai_investment_pool_transfer > 0
            lroai_construction_expenses_of_private_queue >= lroai_investment_pool_transfer_125_percent

            # If you actually have excess reinvestments, the pool itself is for sure not empty
            investment_pool >= lroai_construction_expenses_of_private_queue
        }

        # Equals 1.30x of transfer when it's 1.75x of transfer
        multiply = {
            value = lroai_construction_expenses_of_private_queue
            divide = lroai_investment_pool_transfer
            subtract = 1
            divide = 2.50
            add = 1
            max = 1.30
        }
    }
}

lroai_construction_expenses_of_private_queue = {
    if = {
        limit = {
            has_game_rule = autonomous_investment_pool
        }
        value = construction_queue_num_queued_private_levels
        multiply = lroai_construction_points_per_building
        multiply = lroai_building_construction_sector_expenses_per_point
    }
    else = {
        value = 0
    }
}
