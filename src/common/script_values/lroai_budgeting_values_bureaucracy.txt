        }
        value = var:lroai_building_government_administration_spending
        floor = yes
        divide = {
            value = var:lroai_building_government_administration_spending
            subtract = {
                value = var:lroai_building_government_administration_spending
                floor = yes
            }
            multiply = 100
        }
    }
    else = {
        value = lroai_country_active_income
        multiply = lroai_building_government_administration_spending_share
        multiply = 1.10
    }
    min = 0
    max = 1000000000 #billion
}
